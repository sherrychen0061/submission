    class Solution {
    public:
    int romanToInt(string s) {
    int nums[128];
    nums['I']=1;
    nums['V']=5;
    nums['X']=10;
    nums['L']=50;
    nums['C']=100;
    nums['D']=500;
    nums['M']=1000;
    int result=0;
    int len = s.length();
    int now=nums[s[0]];
    for(int i=0; i<len -1;i++)
    {
    int next=nums[s[i+1]];
    if(now<next)
    {
    result-=now;
    }
    else
    {
    result+=now;
    }
    now=next;
    }
    return result+now;
    }
    };
